def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    dateStr = None
    if city == 'NYC':
        dateStr = datum['starttime'][:-3]
    elif city == 'Chicago':
        dateStr = datum['starttime']
    else:
        dateStr = datum['Start date']
        
    # 
    weekDay = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday']
    date = datetime.strptime(dateStr, '%m/%d/%Y %H:%M')
    month = date.month
    hour = date.hour
    day_of_week = weekDay[date.weekday()]
    
    return (month, hour, day_of_week)

#  `example_trips`
# 
tests = {'NYC': (1, 0, 'Friday'),
         'Chicago': (3, 23, 'Thursday'),
         'Washington': (3, 22, 'Thursday')}

for city in tests:
    assert time_of_trip(example_trips[city], city) == tests[city]

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)
        
            
        # 
        for row in trip_reader:
            # 
            new_point = {}

            ## ##
            ##  new_point  ##
            ##  DictWriter ##
            new_point['duration'] = duration_in_mins(row, city)
            new_point['month'] = time_of_trip(row, city)[0]
            new_point['hour'] = time_of_trip(row, city)[1]
            new_point['day_of_week'] = time_of_trip(row, city)[2]
            new_point['user_type'] = type_of_user(row, city)
            

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)

def number_of_trips(filename):
    """
    
    ,,
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
            else:
                n_customers += 1
        
        # 
        n_total = n_subscribers + n_customers
        
        # 
        return(n_total, n_subscribers/n_total, n_customers/n_total)

def duration_of_trips(filename):
    '''
    
    30
    
    '''
    with open(filename,'r') as f_in:
        reader = csv.DictReader(f_in)
        
        # 
        subcriber_count = 0
        subcriber_duration = 0
        constumer_count = 0
        constumer_duration = 0
        beyond_30min = 0
        
        # 
        for row in reader:
            duration = float(row['duration'])
            
            # 
            if row['user_type'] == 'Subscriber':
                subcriber_count += 1
                subcriber_duration += duration
            else:
                constumer_count += 1
                constumer_duration += duration
                
             # 30   
            if duration > 30:
                beyond_30min += 1
                
        total_duration = subcriber_duration+constumer_duration
        total_count = subcriber_count+constumer_count
        
        return total_duration/total_count,\
                beyond_30min/total_count,\
                subcriber_duration/subcriber_count,\
                constumer_duration/constumer_count

def times_of_everymonth(filename):
    '''
        
        
    '''
    with open(filename, 'r') as f:
        reader = csv.DictReader(f)
        
        subscriber = []
        constumer = []
        for _ in range(12):
            subscriber.append(0)
            constumer.append(0)
        
        for row in reader:
            if row['user_type'] == 'Subscriber':
                subscriber[int(row['month'])-1] += 1
            else:
                constumer[int(row['month'])-1] += 1
                
        return subscriber, constumer
